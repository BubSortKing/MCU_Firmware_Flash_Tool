uint32_t crc32(const uint8_t *data, size_t length)
{
    uint32_t crc = 0xFFFFFFFF;
    for (size_t i = 0; i < length; i++)
    {
        crc = crc ^ data[i];
        for (int j = 0; j < 8; j++) 
        {
            if (crc & 1) 
            {
                crc = (crc >> 1) ^ 0xEDB88320;
            } 
            else 
            {
                crc = crc >> 1;
            }
        }
    }
    return crc ^ 0xFFFFFFFF;
}
